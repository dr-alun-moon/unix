\begin{verbatim}
#!/usr/bin/bash
getopts d withdot && shift
extension=$(echo $name | grep -o '\.[^.]*$')
case $withdot in
  h)
    echo "Usage: extname [-d] <name> | -h"
    echo "  -h show help and exit"
    echo "  -d show extension with leading dot"
    exho "  <name> filename to process"
  d)
    echo $extension
    ;;
  ?)
    echo $extension | tr -d '.'
    ;;
esac
\end{verbatim}
